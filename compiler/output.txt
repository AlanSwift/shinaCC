int binarySearch(int *a, int len, int key)
{
	int left, right, mid;
	left = 0, right = len - 1;
	while(left <= right){
		mid = (left + right) / 2;
		if(a[mid] == key)
			return mid;
		else if(a[mid] < key)
			left = mid + 1;
		else
			right = mid - 1;
	}
	return -1;
}
